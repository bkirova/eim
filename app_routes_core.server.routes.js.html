'use strict';

// Associated controller
var controller = require('../../app/controllers/core');

// Helper modules
var config = require('../../config/config');

/**
 * Core routes module
 * @module app/routes/core.server.routes
 * @param app The ExpressJS app
 */
module.exports = function(app) {

    // GET the index page
    app.get('/', controller.index);

    /**
     * Node environment route (/api/nodeenv). This route returns a JSON object.
     *
     * @example This is an example.
     *
     * @name SomeName
     */
    // GET the Node environment
    app.get('/api/nodeenv', function(req, res) {
        return res.json({ env: process.env.NODE_ENV });
    });

    /**
     * Custom configuration route (/api/config). This route returns the customConfiguration object defined in /config/custom.js.
     *
     * @name SomeName
     */
    app.get('/api/config', function(req, res) {
        return res.status(200).json(config.customConfiguration);
    });
};
